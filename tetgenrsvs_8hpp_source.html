250 double tetgen::internal::ProjectRay(int count, const tetgen::dombounds &boundBox, const T &dir, const V &orig,
251  double minDist)
252 {
253  /*
254  Calculates the distance to project a single ray to reach the bounding box
255 
256  It is templated to accept any types of containers with `count` elements,
257  accessible by operator[];
258  */
259 
260  double l = -INFINITY;
261 
262  for (int i = 0; i < count; ++i)
263  {
264  l = std::max(l, std::min((boundBox[0][i] - orig[i]) / dir[i], (boundBox[1][i] - orig[i]) / dir[i]));
265  }
266  return std::min(l, minDist);
267 }
